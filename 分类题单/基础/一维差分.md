# LC2848. 与车相交的点
[传送门](https://leetcode.cn/problems/points-that-intersect-with-cars/description/)
```C++
class Solution {
public:
    int numberOfPoints(vector<vector<int>>& nums) {
        int diff[102]{};
        for (auto& p : nums) {
            ++ diff[p[0]];
            -- diff[p[1] + 1];
        }
        int ans = 0, s = 0;
        for (auto& d : diff) {
            s += d;
            ans += s > 0;
        }
        return ans;
    }
};
```
