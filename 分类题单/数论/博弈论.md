**巴什博弈**

+ 巴什博弈的游戏规则：有一堆石子，数量为n颗。两个玩家轮流拿石子，甲先取，乙后取，每次可以拿1~m颗石子，轮流拿下去，拿到最后一颗的人获胜。（巴什博弈的特征：只有一堆石子，会给定一个整数m，每次都是从这一堆石子中取1~m颗石子）

**尼姆博弈**

巴什博弈只有一堆石子，如果扩展到多堆石子，情况将复杂很多，这就是尼姆博弈。

+ 尼姆博弈的游戏规则：有n堆石子，数量分别为a_1, a_2, ..., a_n。两个玩家轮流拿石子，甲先取，乙后取，每次从任意一堆中拿走任意数量的石子，轮流拿下去，拿到最后一个石子的玩家获胜。（尼姆游戏的特征：有n堆石子，不会给定整数m，每次都是从任意一堆中拿走任意数量的石子）


# HDU1846 Brave Game
[传送门](https://acm.hdu.edu.cn/showproblem.php?pid=1846)

我们可以先打表找规律，下面是使用SG函数来打表找规律：

```C++
#include <bits/stdc++.h>
using namespace std;
const int N = 1e5 + 10;
int f[N], n, m;

int dfs(int x) {
    if (~f[x]) return f[x];

    unordered_set<int> S;
    for (int i = 1; i <= m; ++ i)
        if (x >= i)
            S.insert(dfs(x - i));

    for (int i = 0; ; ++ i)
        if (!S.count(i))
            return f[x] = i;
}

int main() {
    memset(f, -1, sizeof f);
    n = 50, m = 6;
    for (int i = 0; i <= n; ++ i) {
        printf("SG[%d] = %d\n", i, dfs(i));
    }
}
```

# AcWing892 台阶-Nim游戏
[传送门](https://www.acwing.com/problem/content/894/)


我们使用SG函数打表找规律：

```C++
#include <bits/stdc++.h>
using namespace std;
const int N = 1e5 + 10; // 设置一个足够大的值以适应问题
int a[N], f[N], n;

int dfs(int x) {	// x枚举的是第x级台阶，而不是这个台阶上的石子个数
    if (~f[x]) return f[x];

    unordered_set<int> S;
    for (int i = 1; i <= a[x]; ++ i) // 从第x级台阶可以取1至a[x]个石子
        if (x + 1 < n) // 只有当x不是最后一级台阶时
            S.insert(dfs(x + 1)); // 计算将石子移到下一级后的SG值

    for (int i = 0; ; ++ i)
        if (!S.count(i))
            return f[x] = i; // 找到最小的没有出现的SG值
}

int main() {
    memset(f, -1, sizeof f);
    scanf("%d", &n);
    for (int i = 1; i <= n; ++ i)   scanf("%d", &a[i]);
    for (int i = 1; i < n; ++ i) {
        printf("SG(%d) = %d\n", i, dfs(i));
    }
}
```

# AcWing894. 拆分-Nim游戏
[传送门](https://www.yuque.com/camellia_/zm0aqy/cwoyk8wc3gs7t0q0#pgc7F)
```C++
#include <bits/stdc++.h>
using namespace std;
const int N = 110;
int f[N], n, ans;

int dfs(int x) {
    if (~f[x])  return f[x];
    
    unordered_set<int> S;
    for (int i = 0; i < x; ++ i)
        for (int j = 0; j <= i; ++ j)
            S.insert(dfs(i) ^ dfs(j));
    
    for (int i = 0; ; ++ i)
        if (!S.count(i))
            return f[x] = i;
}

int main() {
    memset(f, -1, sizeof f);
    scanf("%d", &n);
    while (n -- ) {
        int x;
        scanf("%d", &x);
        ans ^= dfs(x);
    }
    puts(ans ? "Yes" : "No");
}
```

# P4279 小约翰的游戏
[传送门](https://www.luogu.com.cn/problem/P4279)
```C++
#include <bits/stdc++.h>
using namespace std;
int n, T, x;

int main() {
    scanf("%d", &T);
    while (T -- ) {
        scanf("%d", &n);
        int ans = 0, sum = 0;
        for (int i = 0; i < n; ++ i) {
            scanf("%d", &x);
            ans ^= x;
            sum += x;
        }
        // 首先考虑最简单的情况,所有石子数都为1,那么奇数堆石子为必败,偶数为必胜
        if (sum == n) {
            puts(n & 1 ? "Brother" : "John");
        } else {
            puts(ans ? "John" : "Brother");
        }
    }
}
```

# P2148 E&D
[传送门](https://www.luogu.com.cn/problem/P2148)

先打表找规律：

```C++
#include<bits/stdc++.h>
using namespace std;
const int N = 10010;
int f[N][N], n;

int dfs(int i, int j) {
    if (~f[i][j])   return f[i][j];

    unordered_set<int> S;
    for (int k = 1; k < i; ++ k)
        S.insert(dfs(k, i - k));
    for (int k = 1; k < j; ++ k)
        S.insert(dfs(k, j - k));

    while (S.count(f[i][j]))    ++ f[i][j];

    return f[i][j];
}

int main() {
    memset(f, -1, sizeof f);
    for (int i = 1; i <= 32; ++ i) {
        for (int j = 1; j <= 32; ++ j) {
            // 这里加1主要是为了输出时好观察，不想看到负数
            printf("%d ", dfs(i, j) + 1);
        }
        puts("");
    }
}
```

```C++
    for (int i = 0; i < 32; ++ i)
        if ((x >> i & 1) == 0)
            return i;
    return 0;

        int ans = 0;

    }
}
```

# P3185 分裂游戏
[传送门](https://www.luogu.com.cn/problem/P3185)

```C++
#include <bits/stdc++.h>
using namespace std;
const int N = 1e4 + 10;
int a[N], f[N], n, T;

int dfs(int i) {
    if (~f[i])  return f[i];

    unordered_set<int> S;
    for (int j = i + 1; j <= n; ++ j)
        for (int k = j; k <= n; ++ k)
            S.insert(dfs(j) ^ dfs(k));	// 这里就类似于AcWing894了

    for (int x = 0; ; ++ x)
        if (!S.count(x))
            return f[i] = x;
}

int main() {
    scanf("%d", &T);
    while (T -- ) {
        memset(f, -1, sizeof f);
        scanf("%d", &n);
        for (int i = 1; i <= n; ++ i)   scanf("%d", &a[i]);

        int ans = 0, tot = 0;
        // 每个豆子都是一个独立游戏，把所有豆子的SG值异或起来
        for (int i = 1; i <= n; ++ i)
            if (a[i] & 1)	// 由于如果a[i]是偶数的话，那么根据异或值相同为0，那么偶数的豆子必然异或起来就是0
                ans ^= dfs(i);	// 此时ans就是SG总值

        // 暴力枚举先手开始执行某种操作，从i里面分裂出俩豆子进了j和k中
        for (int i = 1; i <= n; ++ i) {
            for (int j = i + 1; j <= n; ++ j) {
                for (int k = j; k <= n; ++ k) {
                    // 我们需要的是后手面对SG值为0，这样后手才必败。
                    // 如果SG值为0说明当前先手执行的这个操作是对的，先手执行完这个操作后可以使得后手面对必败局面
                    if ((ans ^ dfs(i) ^ dfs(j) ^ dfs(k)) == 0) {
                        ++ tot;
                        if (tot == 1)   printf("%d %d %d\n", i - 1, j - 1, k - 1);
                    }
                }
            }
        }

        if (!tot)   puts("-1 -1 -1");
        printf("%d\n", tot);
    }
}
```
