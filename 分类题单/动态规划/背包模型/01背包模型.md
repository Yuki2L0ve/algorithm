# AcWing2. 01背包问题
[传送门](https://www.acwing.com/problem/content/2/)

```C++
#include <bits/stdc++.h>
using namespace std;
const int N = 1010;
int f[N], n, m, w, v;

int main() {
    scanf("%d%d", &n, &m);
    for (int i = 1; i <= n; ++ i) {
        scanf("%d%d", &w, &v);
        for (int j = m; j >= w; -- j) {
            f[j] = max(f[j], f[j - w] + v);
        }
    }
    printf("%d\n", f[m]);
}
```

# P1466 Subset Sums

```C++
#include <bits/stdc++.h>
using namespace std;
const int N = 810;
using LL = long long ;
LL f[N], n, m;

int main() {
    scanf("%lld", &n);
    LL sum = n * (n + 1) / 2;
    if (sum & 1) {
        puts("0");
        return 0;
    }

    m = sum / 2;
    f[0] = 1;
    for (int i = 1; i <= n; ++ i)
        for (int j = m; j >= i; -- j)
            f[j] += f[j - i];
    
    printf("%lld\n", f[m] / 2);
}
```

# AcWing12. 背包问题求具体方案
[传送门](https://www.acwing.com/problem/content/12/)
```C++
#include <bits/stdc++.h>
using namespace std;
const int N = 1010, M = N;
int f[N][M], w[N], v[N], n, m;

int main() {
    scanf("%d%d", &n, &m);
    for (int i = 1; i <= n; ++ i) {
        scanf("%d%d", &w[i], &v[i]);
    }
    
    for (int i = n; i; -- i) {
        for (int j = 1; j <= m; ++ j) {
            if (j < w[i]) {
                f[i][j] = f[i + 1][j];
            } else {
                f[i][j] = max(f[i + 1][j], f[i + 1][j - w[i]] + v[i]);
            }
        }
    }
    
    for (int i = 1, j = m; i <= n; ++ i) {
        if (j >= w[i] && f[i][j] == f[i + 1][j - w[i]] + v[i]) {
            printf("%d ", i);
            j -= w[i];
        }
    }
}
```

# P1510 精卫填海
[传送门](https://www.luogu.com.cn/problem/P1510)
```C++
#include <bits/stdc++.h>
using namespace std;
const int N = 10010;
int f[N], n, target, m;

int main() {
    scanf("%d%d%d", &target, &n, &m);
    for (int i = 1; i <= n; ++ i) {
        int v, w;  
        scanf("%d%d", &v, &w);
        for (int j = m; j >= w; -- j)  
            f[j] = max(f[j], f[j - w] + v);
    }

    if (f[m] < target) {
        puts("Impossible");
    } else {
        int j = m;
        while (f[j] >= target)   -- j;
        printf("%d\n", m - (j + 1));
    }
}
```

# P1926 小书童
[传送门](https://www.luogu.com.cn/problem/P1926)
```C++
#include <bits/stdc++.h>
using namespace std;
const int N = 1010;
int t[N], w[N], v[N], f[N], n, m, k, r, tim, ans;

int main() {
    scanf("%d%d%d%d", &n, &m, &k, &r);
    for (int i = 1; i <= n; ++ i)   scanf("%d", &t[i]);
    for (int i = 1; i <= m; ++ i)   scanf("%d", &w[i]);
    for (int i = 1; i <= m; ++ i)   scanf("%d", &v[i]);
    
    for (int i = 1; i <= m; ++ i)
        for (int j = r; j >= w[i]; -- j)
            f[j] = max(f[j], f[j - w[i]] + v[i]);
    
    for (int j = 1; j <= r; ++ j) {
        if (f[j] >= k) {
            tim = r - j;
            break;
        }
    }
    
    sort(t + 1, t + 1 + n);
    for (int i = 1; i <= n; ++ i) {
        tim -= t[i];
        if (tim < 0)    break;
        ++ ans;
    }
    
    printf("%d\n", ans);
}
```
