# 加法
```C++
int add(int a, int b) {
    int ans = a;
    while (b) {
        ans = a ^ b;    // a ^ b表示a和b无进位相加的结果
        b = (a & b) << 1;   // (a & b) << 1表示a和b相加时的进位信息
        a = ans;
    }
    return ans;
}
```

int sub(int a, int b) {
    return add(a, add(~b, 1));
}
```

# 乘法
```C++
int add(int a, int b) {
    int ans = a;
    while (b) {
        ans = a ^ b;    // a ^ b表示a和b无进位相加的结果
        b = (a & b) << 1;   // (a & b) << 1表示a和b相加时的进位信息
        a = ans;
    }
    return ans;
}

int mul(int a, int b) {    // 这个也叫做龟速乘，特别适合 a * b % MOD，其中a和b都是特别大的整数
    int ans = 0;
    while (b) {
        if (b & 1)  ans = add(ans, a);
        add(a, a);
        b >>= 1;
    }
    return ans;
}
