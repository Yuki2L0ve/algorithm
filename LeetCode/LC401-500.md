# LC438. 找到字符串中所有字母异位词
[传送门](https://leetcode.cn/problems/find-all-anagrams-in-a-string/description/?envType=study-plan-v2&envId=top-100-liked)
```java
class Solution {
    public List<Integer> findAnagrams(String s, String p) {
        List<Integer> ans = new ArrayList<>();
        int[] map = new int[26];
        int n = s.length(), m = p.length();
        if (n < m)  return ans;

        for (char c : p.toCharArray()) {
            ++ map[c - 'a'];
        }

        for (int l = 0, r = 0; r < n; ++ r) {
            -- map[s.charAt(r) - 'a'];
            while (map[s.charAt(r) - 'a'] < 0) {
                ++ map[s.charAt(l) - 'a'];
                ++ l;
            }
            if (r - l + 1 == m) {
                ans.add(l);
            }
        }

        return ans;
    }
}
```

```java
class Solution {
    boolean check(int[] cnt, int[] cur) {
        for (int i = 0; i < 26; ++ i)
            if (cnt[i] != cur[i])
                return false;
        return true;
    }

    public List<Integer> findAnagrams(String s, String p) {
        List<Integer> ans = new ArrayList<>();
        int n = s.length(), m = p.length();
        if (n < m)  return ans;

        int[] cnt = new int[26];
        int[] cur = new int[26];
        for (char c : p.toCharArray()) {
            ++ cnt[c - 'a'];
        }

        for (int i = 0; i < m; ++ i) {
            ++ cur[s.charAt(i) - 'a'];
        }
        if (check(cnt, cur)) {
            ans.add(0);
        }

        for (int i = m; i < n; ++ i) {
            -- cur[s.charAt(i - m) - 'a'];
            ++ cur[s.charAt(i) - 'a'];
            if (check(cnt, cur)) {
                ans.add(i - m + 1);
            }
        }

        return ans;
    }
}
```
